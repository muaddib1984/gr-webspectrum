Hotkeys

spacebar = pauses display

'f' key = fullscreen mode

'c' key = cycle spectrum color

'ArrowUp/Down' = pan Y axis

'ArrowLeft/Right' = scale Y axis

's' key = increase spectrum display size

'w' key = increase waterfall display size

'+' key = increment averaging

'-' key = decrement averaging

'm' key = toggle max hold
